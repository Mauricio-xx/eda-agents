% Placeholder box for manual screenshots.
% Usage: \screenshotplaceholder{slides/figures/foo.png}{Short description}
\newcommand{\screenshotplaceholder}[2]{%
  \begin{center}
    \IfFileExists{#1}{%
      \includegraphics[width=0.85\textwidth,keepaspectratio]{#1}%
    }{%
      \fbox{\parbox[c][0.4\textheight][c]{0.85\textwidth}{%
        \centering
        \textit{\color{muted}Screenshot placeholder}\\[0.3em]
        {\footnotesize\ttfamily\detokenize{#1}}\\[0.3em]
        \small #2
      }}%
    }
  \end{center}
}
